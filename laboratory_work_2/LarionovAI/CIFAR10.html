import torch
import torch.nn as nn
import torch.nn.functional as F


class ConvCifar10(nn.Module):
    def __init__(self, num_classes: int = 10, classifier_mode = "MLP"):
        super().__init__()
        self.classifier_mode = classifier_mode
        # ----  1: 3x32x32 -> 32x16x16 ----
        self.block1 = nn.Sequential(
            nn.Conv2d(3, 32, kernel_size=3, padding=1),   # 3x32x32 -> 32x32x32
            nn.BatchNorm2d(32),
            nn.ReLU(inplace=True),

            nn.Conv2d(32, 32, kernel_size=3, padding=1),  # 32x32x32 -> 32x32x32
            nn.BatchNorm2d(32),
            nn.ReLU(inplace=True),

            nn.MaxPool2d(2, 2),                           # 32x32x32 -> 32x16x16
            nn.Dropout(0.25),
        )

        # ----  2: 32x16x16 -> 64x8x8 ----
        self.block2 = nn.Sequential(
            nn.Conv2d(32, 64, kernel_size=3, padding=1),  # 32x16x16 -> 64x16x16
            nn.BatchNorm2d(64),
            nn.ReLU(inplace=True),

            nn.Conv2d(64, 64, kernel_size=3, padding=1),  # 64x16x16 -> 64x16x16
            nn.BatchNorm2d(64),
            nn.ReLU(inplace=True),

            nn.MaxPool2d(2, 2),                           # 64x16x16 -> 64x8x8
            nn.Dropout(0.25),
        )

        # ----  3: 64x8x8 -> 128x4x4 ----
        self.block3 = nn.Sequential(
            nn.Conv2d(64, 128, kernel_size=3, padding=1), # 64x8x8 -> 128x8x8
            nn.BatchNorm2d(128),
            nn.ReLU(inplace=True),

            nn.Conv2d(128, 128, kernel_size=3, padding=1),# 128x8x8 -> 128x8x8
            nn.BatchNorm2d(128),
            nn.ReLU(inplace=True),

            nn.MaxPool2d(2, 2),                           # 128x8x8 -> 128x4x4
            nn.Dropout(0.25),
        )

        # ----  ----
        self.avgpool = nn.AdaptiveAvgPool2d((1, 1))

        self.classifier = nn.Sequential(
            nn.Flatten(),                                 # 128x1x1 -> 128
            nn.Linear(128, 128),
            #KANLayer(128, 32, num = 10, save_plot_data= False, grid_range= [-2.,2.]),
            nn.ReLU(inplace=True),
            nn.Dropout(0.5),
            nn.Linear(128, num_classes),
            #KANLayer(32, num_classes, num = 10, save_plot_data= False, grid_range= [-2.,2.])
        )

    def forward(self, x):
        
        x = self.block1(x)   # -> 32x16x16
        x = self.block2(x)   # -> 64x8x8
        x = self.block3(x)   # -> 128x4x4
        x = self.avgpool(x)  # -> 128x1x1
        if self.classifier_mode == "KAN":
            x = self.classifier[0](x)
            x = self.classifier[1](x)[0]  # KANLayer(128, 128)
            x = self.classifier[2](x)  # ReLU
            x = self.classifier[3](x)  # Dropout
            x = self.classifier[4](x)[0]  # KANLayer(128, num_classes)
        elif self.classifier_mode == "MLP":
            x = self.classifier(x)
        return x

import torch
import tqdm

device = "cuda"
model.to(device)
metrics = [[],[],[],[]]

EPOCHS = 30
#--------- training loop -----------
for epoch in range(EPOCHS):
    model.train()
    train_loss_sum = 0.0

    train_pbar = tqdm.tqdm(trainloader, desc=f"Train epoch {epoch+1}/{EPOCHS}")
    for batch_idx, (X, Y) in enumerate(train_pbar):
        X = X.to(device, non_blocking = True)
        Y = Y.to(device, non_blocking = True)

        optim.zero_grad()

        preds = model(X)
        loss = criterion(preds, Y)

        loss.backward()
        optim.step()

        loss_val = loss.item()
        train_loss_sum += loss_val


        train_pbar.set_postfix(loss=loss_val)

    avg_train_loss = train_loss_sum / len(trainloader)
    metrics[0].append(avg_train_loss)
#--------- training loop -----------

#--------- eval loop -----------
    model.eval()
    val_loss_sum = 0.0
    correct = 0
    total = 0

    with torch.no_grad():
        val_pbar = tqdm.tqdm(testloader, desc=f"Eval epoch {epoch+1}/{EPOCHS}")
        for X_val, Y_val in val_pbar:
            X_val = X_val.to(device, non_blocking = True)
            Y_val = Y_val.to(device, non_blocking = True)

            preds_val = model(X_val)
            loss_val = criterion(preds_val, Y_val)

            loss_batch = loss_val.item()
            val_loss_sum += loss_batch
            
            
            _, predicted = torch.max(preds_val, dim=1)
            correct += (predicted == Y_val).sum().item()
            total += Y_val.size(0)

            val_pbar.set_postfix(loss=loss_batch)
    

    avg_val_loss = val_loss_sum / len(testloader)
    val_acc = correct / total if total > 0 else 0.0
    
    scheduler.step(avg_val_loss)
    
    metrics[1].append(avg_val_loss)
    metrics[2].append(val_acc)
    metrics[3].append(optim.param_groups[0]["lr"])
    
    
#--------- eval loop -----------

    print(
        f"epoch {epoch+1}/{EPOCHS} | "
        f"train_loss={avg_train_loss:.4f} | "
        f"val_loss={avg_val_loss:.4f} | "
        f"val_acc={val_acc:.4f} | "
        f"lr = {optim.param_groups[0]["lr"]:.4f}"
    )
    
torch.save(model.state_dict(), "model.pth")

            if lab not in found:

            if len(found) == 10:     
                break
        if len(found) == 10:
            break

    plt.figure(figsize=(12, 5))
